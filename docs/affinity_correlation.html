slider = alt.binding_range(
    min=escape_func_df["effect"].min(),
    max=escape_func_df["effect"].max(),
    step=0.2,
    name="functional effect",
)
selector = alt.selection_single(
    name="SelectorName",
    fields=["effect"],
    bind=slider,
    init= -1.25,
)

slider_1 = alt.binding_range(
    min=escape_func_df["times_seen"].min(),
    max=escape_func_df["times_seen"].max(),
    step=1,
    name="times seen",
)
selector_1 = alt.selection_single(
    name="SelectorName1",
    fields=["times_seen"],
    bind=slider_1,
    init= 3,
)


corr_chart = (
    alt.Chart(escape_func_df)
    .encode(
        x=alt.X(
            "RBD ACE2 affinity",
            title="RBD affinity (yeast)",
            axis=alt.Axis(grid=False),
            scale=alt.Scale(domain=[escape_func_df["RBD ACE2 affinity"].min(),
                                    escape_func_df["RBD ACE2 affinity"].max()],
                            padding=10)
        ),
        y=alt.Y(
            "median_NT50",
            title="media NT50",
            axis=alt.Axis(grid=False),
            scale=alt.Scale(type="log", domain=[escape_func_df["median_NT50"].min(),
                                    escape_func_df["median_NT50"].max()],
                           padding=10),
        ),
        facet=alt.Facet("antibody", columns=4, title=None),
        tooltip=[
            alt.Tooltip(c, format=".3g") if escape_func_df[c].dtypes == float
            else c
            for c in escape_func_df.columns.tolist()
        ]
    )
    .mark_circle(filled=True, size=60, opacity=0.4)
    .configure_axis(grid=False)
    .resolve_scale(y="independent", x="independent")
    .properties(width=500, height=500)
    .add_selection(selector, selector_1)
    .transform_filter(alt.datum["effect"] > selector["effect"])
    .transform_filter(alt.datum["times_seen"] > selector_1["times_seen"])
)
